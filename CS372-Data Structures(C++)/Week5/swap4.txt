//--------------
// Interface for functin swap
// Given: x and y -- ElementType locations to be swapped
// Return: x and y -- data in locations x and y have been exchanged
//
// Return as value of function: void

#include <cstdio>
#include <iostream>
#include <cstring>

using namespace std;

template <typename ElementType>
void swap( ElementType &x, ElementType &y)
{
	ElementType temp;

	temp = x;
	x = y;
	y = temp;
}

int main()
{
	int a,b;
	a = 2;
	b = 3;
	cout << " a " << a << " b " << b << endl;
	swap( a, b );
	cout << "after swap" << endl;
	cout << " a " << a << " b " << b << endl;

	float r,t;
	r = 2;
	t = 3;
	cout << " r " << r << " t " << t << endl;
	swap( r, t );
	cout << "after swap" << endl;
	cout << " r " << r << " t " << t << endl;

	double x,y;
	x=12345678.0;
	y=99999999.0;
	cout << " x " << x << " y " << y << endl;
	swap( x, y );
	cout << "after swap" << endl;
	cout << " x " << x << " y " << y << endl;

	string s1,s2;
	s1 = new char[20];
	s2 = new char[20];
	s1="String 1";
	s2="String 2";
	cout << " s1 " << s1 << " s2 " << s2 << endl;
	swap( s1, s2 );
	cout << "after swap" << endl;
	cout << " s1 " << s1 << " s2 " << s2 << endl;
	
	return 0;

}
